def xr_all?
  if self.is_a?(Hash)
    xr_each do |key, value|
      return false unless yield(key, value)
    end

      return false unless yield member
    end
  end

end

def xr_each
  if self.is_a?(Hash)
    for key in keys
      yield(key, self[key])
    end
  else
    for i in 0...size
      yield self[i]
    end
  end
end
